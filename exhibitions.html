function getAvailableSlides(gallery) { return Array.from(gallery.querySelectorAll(".slide")).filter( (slide) => slide.dataset.unavailable !== "true" ); } function showSlide(gallery, nextIndex) { const allSlides = Array.from(gallery.querySelectorAll(".slide")); const slides = getAvailableSlides(gallery); const state = galleryState.get(gallery); if (!slides.length) { state.index = 0; state.currentSlide = null; allSlides.forEach((slide) => { slide.classList.remove("is-active"); slide.setAttribute("aria-hidden", "true"); }); state.thumbnails.forEach(({ button }) => { button.classList.remove("is-active"); button.setAttribute("aria-pressed", "false"); }); return; } state.index = (nextIndex + slides.length) % slides.length; state.currentSlide = slides[state.index]; allSlides.forEach((slide) => { const isActive = slide === state.currentSlide; slide.classList.toggle("is-active", isActive); slide.setAttribute("aria-hidden", String(!isActive)); }); state.thumbnails.forEach(({ button, slide }) => { const isActive = slide === state.currentSlide; button.classList.toggle("is-active", isActive); button.setAttribute("aria-pressed", String(isActive)); }); } function nextSlide(gallery) { const state = galleryState.get(gallery); showSlide(gallery, state.index + 1); } function previousSlide(gallery) { const state = galleryState.get(gallery); showSlide(gallery, state.index - 1); } function stopGallery(gallery) { const state = galleryState.get(gallery); if (state.timer) { window.clearInterval(state.timer); state.timer = null; } } function startGallery(gallery) { const state = galleryState.get(gallery); const slideCount = getAvailableSlides(gallery).length; stopGallery(gallery); if ( prefersReducedMotion || state.userPaused || slideCount < 2 || document.hidden ) { return; } state.timer = window.setInterval(() => { nextSlide(gallery); }, 3500); } function chooseActiveGallery() { let bestGallery = null; let bestRatio = 0; galleries.forEach((gallery) => { const ratio = visibility.get(gallery) || 0; if (ratio > bestRatio) { bestRatio = ratio; bestGallery = gallery; } }); galleries.forEach(stopGallery); activeGallery = bestRatio >= 0.5 ? bestGallery : null; if (activeGallery) { startGallery(activeGallery); } } const observer = new IntersectionObserver((entries) => { entries.forEach((entry) => { visibility.set(entry.target, entry.intersectionRatio); }); chooseActiveGallery(); }, { threshold: [0, 0.25, 0.5, 0.75, 1] }); galleries.forEach((gallery) => { const state = galleryState.get(gallery); observer.observe(gallery); gallery.addEventListener("click", () => { if (state.ignoreClick) { return; } state.userPaused = true; stopGallery(gallery); nextSlide(gallery); }); gallery.addEventListener("keydown", (event) => { if (event.key === "ArrowRight" || event.key === "Enter" || event.key === " ") { event.preventDefault(); state.userPaused = true; stopGallery(gallery); nextSlide(gallery); } if (event.key === "ArrowLeft") { event.preventDefault(); state.userPaused = true; stopGallery(gallery); previousSlide(gallery); } }); gallery.addEventListener("touchstart", (event) => { state.touchStartX = event.changedTouches[0].clientX; stopGallery(gallery); }, { passive: true }); gallery.addEventListener("touchend", (event) => { const distance = event.changedTouches[0].clientX - state.touchStartX; if (Math.abs(distance) > 40) { state.ignoreClick = true; state.userPaused = true; if (distance < 0) { nextSlide(gallery); } else { previousSlide(gallery); } window.setTimeout(() => { state.ignoreClick = false; }, 300); } if (!state.userPaused) { chooseActiveGallery(); } }, { passive: true }); }); document.addEventListener("visibilitychange", chooseActiveGallery);
